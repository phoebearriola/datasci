# 3. Design your Cartesian grid and perform a random forest that iterates on each hyperparameter combination.
grid <- expand.grid(
  num.trees = c(100, 300, 500),
  mtry = c(2, 4, 6),
  min.node.size = c(1, 5, 10)
)
grid$OOB_Error <- NA

print(head(grid))

## 3       500    2             1        NA
## 4       100    4             1        NA
## 5       300    4             1        NA
## 6       500    4             1        NA
set.seed(123)
for(i in 1:nrow(grid)) {
rf_fit <- ranger(
    formula         = Species ~ ., 
    data            = df_clean, 
    num.trees       = grid$num.trees[i],
    mtry            = grid$mtry[i],
    min.node.size   = grid$min.node.size[i],
    probability     = FALSE # Standard classification to get clear OOB error
  )
grid$OOB_Error[i] <- rf_fit$prediction.error
}
print(grid)

# 5. Using the optimal hyperparameter, provide a plot of variable importance and the predicted class probabilities.

## Variable Importance Plot
optimal_rf <- ranger(
  formula         = Species ~ ., 
  data            = df_clean, 
  num.trees       = best_row$num.trees,
  mtry            = best_row$mtry,
  min.node.size   = best_row$min.node.size,
  importance      = "permutation"
)

imp_scores <- importance(optimal_rf)
imp_df <- data.frame(
  Variable = names(imp_scores),
  Importance = as.numeric(imp_scores)
) %>% arrange(desc(Importance))

ggplot(imp_df, aes(x = reorder(Variable, Importance), y = Importance)) +
  geom_bar(stat = "identity", fill = "steelblue") +
  coord_flip() +
  labs(title = "Variable Importance Plot (Optimal Model)",
       x = "Predictors",
       y = "Permutation Importance Score") +
  theme_minimal()

## Predicted Class Probabilities Plot
prob_rf <- ranger(
  formula         = Species ~ ., 
  data            = df_clean, 
  num.trees       = best_row$num.trees,
  mtry            = best_row$mtry,
  min.node.size   = best_row$min.node.size,
  probability     = TRUE
)

predicted_probs <- predict(prob_rf, data = df_clean)$predictions
